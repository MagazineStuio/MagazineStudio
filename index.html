สคริปต์เบื้องต้น (Luau)

เพิ่ม Script ใต้ ServerScriptService หรือ LocalScript ใต้ StarterPlayerScripts

-- พิมพ์ข้อความไปที่ Output
print("สวัสดี Roblox!")

-- ตัวแปรและตาราง
local speed = 16
local playerStats = {coins = 0, level = 1}

-- ฟังก์ชัน
local function reward(player)
  playerStats.coins += 10
  print(player.Name .. " ได้รับ 10 เหรียญ")
end

-- เชื่อมเหตุการณ์เมื่อผู้เล่นเข้ามา
local Players = game:GetService("Players")
Players.PlayerAdded:Connect(function(player)
  print("ยินดีต้อนรับ ", player.Name)
  reward(player)
end)

โต้ตอบกับวัตถุ

-- ทำให้ Part เปลี่ยนสีเมื่อสัมผัส
local part = workspace:WaitForChild("ColorPad")

part.Touched:Connect(function(hit)
  local p = hit.Parent
  if p:FindFirstChild("Humanoid") then
    part.BrickColor = BrickColor.Random()
  end
end)
สร้าง Part ชื่อ ColorPad ใน Workspace ก่อนรัน

แนวทาง UX/UI สำหรับ Roblox

หลักการที่เน้นเล่นง่าย อ่านง่าย และเฟรมเรตเสถียร

การอ่านง่าย
  • ใช้ UIStroke รอบตัวอักษรบนพื้นหลังสว่าง
  • ตั้ง UITextSizeConstraint ให้ปรับตามหน้าจอ
  • ใช้สีคู่คอนทราสต์สูงและระวังสีแดง/เขียวร่วมกัน
ประสิทธิภาพ
  • รวม UI เป็น ScreenGui เดียวเพื่อลด draw calls
  • ใช้งาน TweenService สำหรับอนิเมชั่นที่ลื่น
  • โหลดภาพแบบเชิงรุกและใช้ Asset ที่เหมาะกับมือถือ
การป้อนข้อมูล
  • รองรับคีย์บอร์ด เมาส์ และจอยแพด
  • ปุ่มหลักควรมีขนาดใหญ่พอสำหรับนิ้วโป้ง
  • ให้ Feedback ด้วยเสียง/อนิเมชั่นสั้น ๆ
สไตล์ภาพ
  • โทนสี 2–3 สีหลัก ตั้งเป็น Theme
  • ใช้เงาเบา ๆ และมุมโค้งเพื่อความทันสมัย
  • อย่าแสดง UI เกินจำเป็นบนหน้าจอ
